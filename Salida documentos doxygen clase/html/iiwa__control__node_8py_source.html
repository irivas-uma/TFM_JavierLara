68 def pose_command_callback(self, msg):
69 """
70 Callback para recibir comandos de posición cartesiana.
71 Args:
72 msg: Mensaje de posición cartesiana (PoseStamped).
73 """
74 if self.work_mode == "free":
75 # Mover el robot a una posición cartesiana específica en modo "free"
76 pose = PoseStamped()
77 pose = msg
78 self.iiwa.move_cartesian(pose)
79 elif self.work_mode == "pivot":
80 # Mover el robot en modo "pivot" alrededor de un punto de fulcro
81 pose = PoseStamped()
82 pose = msg
83
84 # Obtener la posición x, y, z del mensaje de entrada
85 position = np.array([msg.pose.position.x, msg.pose.position.y, msg.pose.position.z])
86
87 if self.first_position is None:
88 # Primer mensaje recibido, enviar directamente la posición al método
89 self.first_position = position
90 increment_vector = np.zeros(3) # Crear un vector de ceros
91 j = 0
92 self.iiwa.move_cartesian_fulcrum(pose, increment_vector, j)
93 else:
94 # Calcular vector de incrementos
95 increment_vector = position - self.first_position
96 j = 1
97 # Enviar vector de incrementos y pose al método
98 self.iiwa.move_cartesian_fulcrum(pose, increment_vector, j)
99 else:
100 rospy.logwarn("Modo de trabajo no válido.")

132 def cartesian_pose_callback(self, msg):
133 """
134 Callback para recibir estados de posición cartesiana.
135 Args:
136 msg: Mensaje de estado de posición cartesiana (PoseStamped).
137 """
138 # Publicar la información recibida
139 self.ef_pose_pub.publish(msg)
140
141 # Calcular la posición del TCP a partir del EF
142
143 # Obtener la posición y orientación del mensaje
144 position = msg.pose.position
145 orientation = msg.pose.orientation
146
147 # Reordenar el cuaternio en el vector q (w, x, y, z)
148 q = [orientation.w, orientation.x, orientation.y, orientation.z]
149
150 # Crear una matriz de rotación a partir del cuaternio recibido
151 Rm = matrix_from_quaternion(q)
152
153 # Calcular el vector de dirección relativo al EF
154 direction = np.array([0,0,self.tool_length])
155 direction_transformed = np.dot(Rm, direction)
156
157 # Guardar la posición del EF del robot en la variable Pef
158 Pef = np.array([position.x,position.y,position.z])
159
160 # Calcular la posición del TCP
161 Ptcp = Pef + direction_transformed
162
163 # Actualizar valores de posición del TCP
164 position.x = Ptcp[0]
165 position.y = Ptcp[1]
166 position.z = Ptcp[2]
167
168 # Crear un mensaje del tipo PoseStamped
169 tcp_pose = PoseStamped()
170
171 # Actualizar el mensaje de tcp_pose con los valores ajustados
172 tcp_pose.header = msg.header
173 tcp_pose.pose.position = position
174 tcp_pose.pose.orientation = orientation # La herramienta tendrá la misma orientación que el EF
175
176 # Publicar el mensaje
177 self.tcp_pose_pub.publish(tcp_pose)
